\section{コマンドラインオプション}

\begin{verbatim}
#include <cure/cure_option.h>

zOption option[] = {
  { "x",  "xarg", "<x val>", "x value", NULL, false },
  { "y",    NULL, "<y val>", "y value", NULL, false },
  { NULL, "zarg", "<z val>", "z value", NULL, false },
  { NULL,   NULL,      NULL,      NULL, NULL, false },
};

int main(int argc, char *argv[])
{
  char *arg1, *arg2;

  zOptionHelp( option );
  arg1 = arg2 = NULL;
  zOptionRead( option, argv+1, 2, &arg1, &arg2 );
  printf( "arg1 = %s\n", arg1 ? arg1 : "NULL" );
  printf( "arg2 = %s\n", arg2 ? arg2 : "NULL" );
  zOptionDiag( option );
  return 0;
}
\end{verbatim}
